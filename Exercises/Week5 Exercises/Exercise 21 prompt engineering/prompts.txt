SELECT
  department,
  AVG(salary) AS avg_salary,
  COUNT(*) AS employee_count
FROM
  `my_project.my_dataset.employees`
GROUP BY
  department
HAVING
  COUNT(*) > 10
ORDER BY
  avg_salary DESC
LIMIT 5;

import pandas as pd
from datetime import datetime

def validate_order_data(df: pd.DataFrame) -> dict:
    """
    Validates order data in a pandas DataFrame.

    Checks for:
    - Null values in any column
    - Negative values in 'amount'
    - Future dates in 'order_date'
    - Duplicate values in 'order_id'

    Returns a dictionary with boolean results for each check.
    """
    null_values = df.isnull().any().any()
    negative_amounts = (df["amount"] < 0).any()
    future_dates = (pd.to_datetime(df["order_date"]) > datetime.now()).any()
    duplicate_ids = df["order_id"].duplicated().any()

    return {
        "null_values": bool(null_values),
        "negative_amounts": bool(negative_amounts),
        "future_dates": bool(future_dates),
        "duplicate_ids": bool(duplicate_ids),
    }
